COURSE
_________________________________
drop table course;
CREATE TABLE IF NOT EXISTS course(course_id int(10) primary key auto_increment, 
assessment_id int(10) not null,
foreign key(assessment_id) references assessment(assessment_id),
term_id int(10) not null,
foreign key(term_id) references term(term_id),
mentor_id int(10) not null,
foreign key(mentor_id) references mentor(mentor_id),
title varchar(30) not null,
status boolean not null,
completed boolean,
notes varchar(1000) not null,
start datetime not null,
end datetime not null);
explain course;


TERM
_________________________________
drop table term;
CREATE TABLE IF NOT EXISTS term(term_id int(10) primary key auto_increment,
course_id int(10) not null,
foreign key(course_id) references course(course_id),
title varchar(30) not null,
completed boolean,
start datetime not null,
end datetime not null);
alter table term modify column course_id int(10) NOT NULL, add key(course_id);
explain term;


ASSESSMENT
_________________________________
drop table assessment;
CREATE TABLE IF NOT EXISTS assessment(assessment_id int(10) primary key auto_increment,
course_id int(10) not null,
foreign key(course_id) references course(course_id),
title varchar(30) not null,
date datetime not null);
alter table assessment modify column course_id int(10) NOT NULL, add key(course_id);
explain assessment;


Mentor
_________________________________
drop table mentor;
CREATE TABLE IF NOT EXISTS mentor(mentor_id int(10) primary key auto_increment,
course_id int(10) not null,
foreign key(course_id) references course(course_id),
name varchar(30) not null,
email varchar(30) not null,
phone varchar(30) not null);
alter table mentor modify column course_id int(10) NOT NULL, add key(course_id);
explain mentor;
